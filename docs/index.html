▸ Why crystalsystem.js

Everything you need.
Nothing you don't.

Built entirely on Node.js built-ins. No audit warnings, no dependency drama, no version conflicts.

⚡
Real CPU %
Actual usage via CPU time-delta sampling — not just load average. Far more accurate for spikes.
📊
Sparklines
Live ASCII history graphs for CPU and memory rendered directly inside your terminal dashboard.
🔔
Alert System
Fire callbacks when CPU, memory or disk exceeds configurable thresholds. Built-in cooldown prevents spam.
📡
Event Emitter
cs.on('data', fn) and cs.on('alert', fn) for tight integration with your own systems.
🌐
HTTP API
Built-in JSON server at any port. Wire into Grafana, Discord bots, web dashboards — no setup needed.
📝
File Logging
Append JSON-line stats to a file every tick. Read them back with logger.readAll() for analysis.
🖥️
CLI Binary
crystalsystem --once --json -p 3001 -l stats.json — full control from your terminal.
🔝
Top Processes
Shows top N processes sorted by memory usage inside the dashboard — like a lightweight htop.
📘
TypeScript
Full .d.ts definitions for every function, interface and option. No @types package needed.

▸ Usage

From zero to live monitor
in three lines.

Every example below runs as-is. No config, no boilerplate.

// Just require and go — dashboard starts immediately
const { crystalsystem } = require('@crystal-studio-labs/crystalsystem.js');

crystalsystem({
  interval:      5000,   // refresh every 5s
  clearConsole:  true,
  showColors:    true,
  showNetwork:   true,
  showDisk:      true,
  showBattery:   true,
  showGPU:       true,
  showProcesses: true,
  httpPort:      3001,   // optional live JSON API
  logFile:       'stats.json',
  thresholds: {
    cpu:    80,
    memory: 75,
    disk:   85,
  },
});
const { CrystalSystem } = require('@crystal-studio-labs/crystalsystem.js');

const cs = new CrystalSystem({
  interval:   2000,
  thresholds: { cpu: 80, memory: 75 },
});

// Every tick — full stats object
cs.on('data', (stats) => {
  console.log('CPU:', stats.cpu.loadPercent);
  console.log('RAM:', stats.memory.percent);
  console.log('Net:', stats.network.rxSpeed);
});

// Threshold breach — send webhook, restart, log, anything
cs.on('alert', ({ type, message, value, threshold }) => {
  console.error(`⚠️  [${type}] ${message}`);
});

cs.on('stop', () => console.log('Stopped.'));

cs.start();

// Update threshold at runtime
cs.setThreshold('cpu', 70);

// One-shot snapshot
const snap = cs.snapshot();
console.log(snap.cpu.model, snap.memory.total);
const { crystalsystem } = require('@crystal-studio-labs/crystalsystem.js');

// Start dashboard + HTTP API together
crystalsystem({ httpPort: 3001 });

// Available endpoints:
// GET http://localhost:3001/          → all stats
// GET http://localhost:3001/cpu       → CPU info
// GET http://localhost:3001/memory    → memory info
// GET http://localhost:3001/network   → network info
// GET http://localhost:3001/disk      → disk info
// GET http://localhost:3001/battery   → battery info
// GET http://localhost:3001/health    → { status: 'ok' }

// Or start the server independently
const { startHttpServer } = require('@crystal-studio-labs/crystalsystem.js');
const server = startHttpServer(3001);
// server is a Node.js http.Server
import crystalsystem, {
  CrystalSystem,
  CrystalSystemOptions,
  SystemStats,
  AlertEvent,
  CPUInfo,
} from '@crystal-studio-labs/crystalsystem.js';

const options: CrystalSystemOptions = {
  interval:   3000,
  thresholds: { cpu: 80, memory: 80 },
  httpPort:   3001,
};

const cs: CrystalSystem = crystalsystem(options);

cs.on('data', (stats: SystemStats) => {
  const cpu: CPUInfo = stats.cpu;
  console.log(cpu.loadPercent, stats.memory.percent);
});

cs.on('alert', (alert: AlertEvent) => {
  console.error(`[${alert.type.toUpperCase()}] ${alert.message}`);
});
# Install globally
$ npm install -g @crystal-studio-labs/crystalsystem.js

# Live dashboard (5s refresh)
$ crystalsystem

# Refresh every 2 seconds
$ crystalsystem -i 2

# Show once and exit
$ crystalsystem --once

# JSON snapshot to stdout
$ crystalsystem --json

# Dashboard + HTTP API + file log
$ crystalsystem -p 3001 -l stats.json

# Hide sections
$ crystalsystem --no-disk --no-gpu --no-battery

# Pipe clean output
$ crystalsystem --once --no-color > snapshot.txt

# Help
$ crystalsystem --help
